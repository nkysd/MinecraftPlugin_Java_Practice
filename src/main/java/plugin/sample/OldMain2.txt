package plugin.sample;

import java.util.Arrays;
import java.util.List;
import java.util.Objects;
import org.bukkit.Bukkit;
import org.bukkit.Color;
import org.bukkit.FireworkEffect;
import org.bukkit.FireworkEffect.Type;
import org.bukkit.World;
import org.bukkit.entity.Firework;
import org.bukkit.entity.Player;
import org.bukkit.event.EventHandler;
import org.bukkit.event.Listener;
import org.bukkit.event.player.PlayerBedEnterEvent;
import org.bukkit.event.player.PlayerToggleSneakEvent;
import org.bukkit.inventory.ItemStack;
import org.bukkit.inventory.meta.FireworkMeta;
import org.bukkit.plugin.java.JavaPlugin;

public final class OldMain2 extends JavaPlugin implements Listener {

  // プレーヤーがスニークした回数をカウントするための変数
  private int count;

  @Override
  public void onEnable() {
    Bukkit.getPluginManager().registerEvents(this, this);

  }

  /**
   * プレイヤーがスニークを開始/終了する際に起動されるイベントハンドラ。
   *
   * @param e イベント
   */
  @EventHandler
  public void onPlayerToggleSneak(PlayerToggleSneakEvent e) {
    // イベント発生時のプレイヤーやワールドなどの情報を変数に持つ。
    Player player = e.getPlayer();
    World world = player.getWorld();

    // 花火のカラーリスト
    List<Color> colorList = List.of(Color.RED, Color.BLUE, Color.WHITE, Color.BLACK);

    // プレーヤーがスニークしてる時だけ(スニーク回数が偶数の時(割る2して余りが0のとき))
    if (count % 2 == 0) {
      // カラーリストをfor文で回す
      for (Color color : colorList) {

        // 花火オブジェクトをプレイヤーのロケーション地点に対して出現させる。
        Firework firework = world.spawn(player.getLocation(), Firework.class);

        // 花火オブジェクトが持つメタ情報を取得。
        FireworkMeta fireworkMeta = firework.getFireworkMeta();

        // メタ情報に対して設定を追加したり、値の上書きを行う。
        // 今回は青色で星型の花火を打ち上げる。 -> カラーリストを使う
        fireworkMeta.addEffect(
            FireworkEffect.builder()
                // .withColor(Color.BLUE)
                .withColor(color)
                .with(Type.STAR)
                .withFlicker()
                .build());
        fireworkMeta.setPower(0);

        // 追加した情報で再設定する。
        firework.setFireworkMeta(fireworkMeta);

        // 入出力処理
        // テキストファイルを開いて(なければ作成して)、文字列を書き込む(毎回上書きされる)
        Path path = Path.of("firwork.txt");
        Files.writeString(path, "たーまやー");
        //文字列を読み込んで、メッセージとして送る
        player.sendMessage(Files.readString(path));
      }
    }
      // ifの分岐がどうだろうと、処理が終わったらcountに+1する
      count++;
  }

  @EventHandler
  // プレイヤーがベッドに入ろうとしたときに、すべてのスタックできるアイテムの個数を64(Max)にする
  public void onPlayerBedEnter(PlayerBedEnterEvent e){
    Player player = e.getPlayer();
    ItemStack[] itemStacks = player.getInventory().getContents(); // inventory(アイテム)
    // アイテムがnullではなく
    // スタックできるアイテムのマックス数が64(デフォルト値)で
    // かつ持ってるのが64以下のとき
    Arrays.stream(itemStacks).forEach(item -> {
      if (!Objects.isNull(item) && item.getMaxStackSize() == 64 && item.getAmount() < 64) {
        // それを64こにする
        item.setAmount(64);
      }
      // 64こになったアイテムを上書き
      player.getInventory().setContents(itemStacks);
    });


  }
}
